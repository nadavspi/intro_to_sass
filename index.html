Quick sidenote

Beware of specificity

&

parent selector

Sass 
              
a {
  color: saddlebrown;

  &:hover {
    color: wheat;
  }
}
CSS 
                
a {
  color: saddlebrown;
}

a:hover {
  color: wheat;
}
Sass 
                        
a {
  text-decoration: none;

  p & {
    text-decoration: underline;
  }
}
CSS 
                          
a {
  text-decoration: none;
}

p a {
  text-decoration: underline;
}

Variables

                        
// Colors
$pink: #e20177;
$blue: #75c3c8;

// Font stacks
$sans: "HelveticaNeue-Light", "Helvetica Neue Light",
            "Helvetica Neue", Helvetica, Arial, "Lucida Grande", sans-serif;
$serif: "ff-spinoza-web-pro", Times, Georgia, serif;
Sass 

/* Headings */
h1 {
  font-family: $sans;
  color: $pink;
}

// I like this one more
h2 {
  font-family: $serif;
  color: $blue;
}
CSS 

/* Headings */
h1 {
  font-family: "HelveticaNeue-Light", "Helvetica Neue Light", "Helvetica Neue", Helvetica, Arial, "Lucida Grande", sans-serif;
  color: #e20177;
}

h2 {
  font-family: "ff-spinoza-web-pro", Times, Georgia, serif;
  color: #75c3c8;
}

@import

Already exists in CSS, but improved upon

_headings.scss 

h1 {
  font-size: 2em;
  line-height: 1.2;
}
_buttons.scss 

.button {
  padding: .5em 1em;
  text-transform: uppercase;
  background-color: deepskyblue;
}
styles.scss 

@import "headings";
@import "buttons";
